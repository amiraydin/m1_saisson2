    val Array(salaire, devise) = salaireDevise.split(" ");
    val salaireEnEuros = if (devise == "$") salaire.toDouble / 2 else salaire.toDouble;
    if (salaireEnEuros <= 2000) {println(s"$nom - $salaireEnEuros?")}
}

var salairesModestes = 0
var salairesMoyens = 0
var salairesConfortables = 0

listSal.foreach { salaireString =>
    val Array(nom, entreprise, salaireDevise) = salaireString.split(" - ")
    val Array(salaire, devise) = salaireDevise.split(" ")

    if (salaireEnEuros < 2000) {

    salairesMoyens += 1
